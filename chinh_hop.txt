#define MAX 100
using namespace std;

std::ifstream input("input.txt");
std::ofstream output("output.txt");

int n,k;
int x[MAX];
int mark[MAX];
string dict[MAX];
void xuat_dict(){
    for(int i = 1;i<=k;i++){
        output<<dict[x[i]];
    }
    output<<"\n";
}
void input_dict(){
     for(int i = 1;i<=k;i++){
        dict[i]=to_string(x[i]);
    }
}
void hoan_vi(int i){
    for(int v=1;v<=k;v++){
        if(mark[v]==false){
            x[i]=v;
            mark[v]=true;
            if(i==k){
               xuat_dict();
            }
            else hoan_vi(i+1);
            mark[v]=false; // â† backtrack step happens here
                            //sau khi goi de quy xong thi x[2] unmark cai mark[v(2)] de su
        }

    }
}

void to_hop(int i){
    for(int v=x[i-1]+1;v<=n-k+i;v++){
        if(true){
            x[i]=v;
            if(i==k){
                input_dict();
                hoan_vi(1);
            }
            else to_hop(i+1);
        }

    }
}
void solution(){

}
