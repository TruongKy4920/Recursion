std::ifstream input("input.txt");

int n,m;
int x[MAX];


void xuat(int tren, int duoi, bool upper_first){
    bool have_open=false;
     for(int i=1;i<=tren;i++){
        if(x[i]=='O') have_open=true;
     }
     if(!have_open) return; //có thể ghi if ( have_open == false)

     if(upper_first){
        for(int i=1;i<=n;i++)  output<<char(x[i])<<" ";            
        for(int i=1;i<=m;i++)  output<<"C ";
        output<<"\n";
     }
     else{
        for(int i=1;i<=m;i++) output<<char(x[i])<<" ";
        for(int i=1;i<=n;i++) output<<"C ";
        output<<"\n";
     }

}
void cong_tacdien(int i,int tren,int duoi,bool upper_first){
    for(int v : {'C', 'O'}){
        if(true){
            x[i]=v;
            if(i==tren) {
                xuat(tren,duoi,upper_first);               
            }
            else cong_tacdien(i+1,tren,duoi,upper_first);
        }
    }
}

}
