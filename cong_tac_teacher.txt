#include <iostream>
#include <fstream>
#include <string>
#include <sstream>

#define MAX 100
std::ifstream input("input.txt");
std::ofstream output("output.txt"); // ğŸ” Replace cout with output

int x[MAX];
int count = 0;

void print_result(int n, int m, bool upper_first) {
    bool have_open = false;
    for(int i = 1; i <= n; i++) {
        if(x[i] == 'O') have_open = true;
    }
    if(!have_open) return;
    
    count++;
    if(upper_first){
        for (int i = 1; i <= n; i++) output << char(x[i]) << " ";
        for(int i = 1; i <= m; i++) output << "C ";
    }
    else {
        for(int i = 1; i <= m; i++) output << "C ";
        for (int i = 1; i <= n; i++) output << char(x[i]) << " ";
    }
    output << "\n";
}

void resolve(int i, int n, int m, bool upper_first) {
    for(int v : {'C', 'O'}) {
        x[i] = v;
        if (i == n) {
            print_result(n, m, upper_first);
        }
        else {
            resolve(i + 1, n, m, upper_first);
        }
    }
}

int main() {
    int n, m;
    input >> n >> m;
    
    for(int i = 1; i <= m + n; i++) output << "C ";
    output << "\n";
    
    resolve(1, n, m, true);
    resolve(1, m, n, false);
    
    output << count + 1;
}
