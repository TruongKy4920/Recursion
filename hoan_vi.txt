#include <iostream>

#include <math.h>
#define MAX 100
using namespace std;

std::ifstream input("input.txt");
std::ofstream output("output.txt");

int n,k;
int x[MAX];
bool mark[MAX];
void xuat(){
    for(int i = 1;i<=n;i++){
        output<<x[i];
    }
    output<<"\n";
}
void hoan_vi(int i){
    for(int v=1;v<=n;v++){
        if(mark[v]==false){
            x[i]=v;
            mark[v]=true;
            if(i==n){
                xuat();
            }
            else hoan_vi(i+1);
            mark[v]=false; // â† backtrack step happens here
                            //sau khi goi de quy xong thi x[2] unmark cai mark[v(2)] de su
        }

    }
}
void solution(){
   hoan_vi(1);
}

int main(){
    std::string line;
      input>>n>>k;
      solution();
